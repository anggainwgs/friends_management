Resource Description
POST /accounts/connect

#1 create a friend connection between two email addresses

params example:

{ friends: ['andy@example.com', 'john@example.com'] }

success response:

{ "success": true }

errors response:

{ "error": message, "status": 401 }
GET /accounts

#2 show friends list for an email address.

params example:

{ "email": 'andy@example.com', 'john@example.com' }

success response:

{ "success": true, "friends": ['john@example.com', 'common@example.com'],"count": 2}

errors response:

{ "error": message, "status": 401 }
GET /accounts/common_friends

#3 retrieve the common friends list between two email addresses.

params example:

{ "emails": ['andy@example.com', 'john@example.com'] }

success response:

{ "success": true, "friends": ['common@example.com'] }

errors response:

{ "error": message, "status": 401 }
POST /accounts/add_subscribe

#4 subscribe to updates from an email address.

params example:

{ "requestor": 'andy@example.com', "target": 'john@example.com' }

success response:

{ "success": true }

errors response:

{ "error": message, "status": 401 }
POST /accounts/block_subscribe

#5 to block updates from an email address.

params example:

{ "requestor": 'andy@example.com', "target": 'john@example.com' }

success response:

{ "success": true }

errors response:

{ "error": message, "status": 401 }
POST /accounts/add_update

#6 retrieve all email addresses that can receive updates from an email address.

params example:

{ "sender": 'andy@example.com', "text": 'hello john@example.com' }

success response:

{ "success": true, "recipients": ['john@example.com','common@example.com'] }

errors response:

{ "error": message, "status": 401 }
